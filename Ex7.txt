#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() {
    char input[100];
    char litere[4];
    int numere[4];

    printf("Introduceti 4 litere si 4 numere intregi alternativ: ");
    gets_s(input);

    int litereIndex = 0;
    int numereIndex = 0;

    for (int i = 0; input[i] !=0; i++) {
        char ch = input[i];
        if ((ch >= 'A' && ch <= 'Z') || (ch >= 'a' && ch <= 'z')) {
            if (litereIndex < 4) {
                litere[litereIndex++] = ch;
            }
        }
        else if (ch >= '0' && ch <= '9') {
            if (numereIndex < 4) {
                numere[numereIndex++] = ch - '0';
            }
        }
    }

    printf("Afisare: ");
    for (int i = 3; i >= 0; i--) {
        printf("%c%d", litere[i], numere[i]);
    }
    printf("\n");

    return 0;
}
